Goals

  • cross platform
  • data-intensive applications
  • strongly typed
  • snappy & responsive
We believe it's possible to strongly reduce tedious and repeatable code. Prime.vc: A series of light-weight, well documented libraries for developing data-driven applications. Our libraries are designed to be: * cross platform (flash9 all, js some) * avoid library lockin functionality should be replacable by a different library * fast performed lot of benchmarking to find out the fastest procedures (depending on platform)

Signals

  • Strongly-typed event-system
  • Intuitive syntax-sugar
  • Cross-platform

What's wrong with events?

  • no typechecking
  • messages wrapped in Event-class
  • no generic functions as listeners
var signal : Signal2<Int, String>                   // signal definition

function actOnSignal (p1:Int, p2:String) : Void;    // handler definition

signal.bind( this, actOnSignal );                   // binding them together

Binding, observing


import prime.signals.Signal2;

class Test {
	static function main() new Test()
	function new () {
		var s = new Signal2<Int,String>();
		s.bind(this, function (a,b) trace("a: "+a+"; b: "+b));
		s.observe(this, function () { trace("test"); });
		s.send(5, "test");
	}
}

Syntax sugar


import prime.signals.Signal1;
 using prime.utils.Bind;
class Test {
	static function main() new Test()
	function new () {
		var s = new Signal1<String>();
		function () { trace("test"); }.on(s,this);
		function () { trace("test once"); }.onceOn(s,this);
		function (p){ trace(p); }.on(s,this);
		function (p){ trace(p+" once"); }.onceOn(s,this);
		s.send("argument1");
		s.send("argument2");
	}
}

Bindable Object


import prime.bindable.Bindable;
 using prime.utils.Bind;
class Test {
	static function main () new Test()
	function new () {
		var a = new Bindable<String>("WWX '13");
		function (newV, oldV) {
			trace("a changed from "+oldV+" => "+newV);
		}.on(a.change, this);
		trace(a.value);
		a.value = "Enjoy WWX '13";
	}
}

Binding Objects


import prime.bindable.Bindable;
class Test {
	static function main () new Test()
	function new () {
		var a = new Bindable<String>("Value A");
		var b = new Bindable<String>("Value B");
		trace("A: "+a.value);
		trace("B: "+b.value+"\n");
		b.bind(a);

		trace("change A");
		a.value = "Enjoy WWX '13";
		trace("A: "+a.value);
		trace("B: "+b.value+"\n");

		trace("change B");
		b.value = "Back to value B";
		trace("A: "+a.value);
		trace("B: "+b.value);
	}
}

Adding children


import prime.gui.events.MouseEvents;
 using prime.utils.Bind;

class Test extends prime.gui.display.Sprite {
	static function main() {
		var w = new prime.gui.display.Window(flash.Lib.current.stage);
		w.children.add(new Test());
	}

	public function new (x = 0.0, y = 0.0) {
		super();
		this.x = x; this.y = y;
		funkify(prime.utils.Color.random());
		changeColor .on( userEvents.mouse.rollOver, this );
		changeColor .on( userEvents.mouse.rollOut,  this );
		addUglyChild.on( userEvents.mouse.click,    this );
	}

	function changeColor (state:MouseState)
		funkify(Std.int( 0xF1F100 * state.local.x ))

	function addUglyChild () {
		var x = container.children.length * 15;
		children.add(new Test(x, parent.y + 37.5));
	}

	function funkify (color) {
		graphics.clear();
		graphics.beginFill(color);
		graphics.drawRect(0,0,75,75);
	}
}

LayoutClients

Example LayoutClient


import prime.layout.LayoutClient;
import prime.gui.display.Sprite;
import prime.core.geom.Box;
 using prime.utils.Bind;

class Test {
	static var sprite = new Sprite();
	static var layout = new LayoutClient(100, 50);
	static function main () {
		layout.margin = new Box(10);
		flash.Lib.current.stage.addChild(sprite);
		redraw.on(layout.changed, sprite);
		move.on(sprite.userEvents.mouse.click, sprite);
		redraw();
	}

	static function move () layout.x += 10

	static function redraw () {
	trace("redraw");
		var g = sprite.graphics;
		var l = layout.innerBounds;
		g.clear();
		g.beginFill(0xff0000);
		g.drawRect(l.left, l.top, l.width, l.height);
		g.endFill();
	}
}

Example Algorithms


import Prime;
 using prime.utils.Bind;
 using Std;

class Test extends prime.gui.core.UIWindow
{
	public static function main ()
		Window.startup(function (stage) return new Test(stage))

	override private function createChildren () {
		for (i in 0...15)	attach(new UIGraphic());
		changeAlgorithm.on(userEvents.mouse.click, this);
	}

	function changeAlgorithm ()
		layoutContainer.algorithm = cast a[a.length.random()]

	static var a = [
		new HorizontalFloat(left,   top),
		new HorizontalFloat(center, bottom),
		new VerticalFloat(top,      right),
		new VerticalFloat(bottom,   left),
		new SimpleTile(),
		new SimpleTile(vertical),
		new DynamicLayout(
			function () return new HorizontalCircle(),
			function () return new VerticalCircle()),
		new DynamicLayout(
			function () return new HorizontalCircle(),
			function () return new VerticalFloat())];
}
